<?php
require_once 'db-init.php';

// Обработка создания опроса
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $title = $_POST['title'] ?? '';
    $description = $_POST['description'] ?? '';
    
    if (!empty($title)) {
        // Создаем опрос
        $stmt = $pdo->prepare("INSERT INTO surveys (title, description) VALUES (?, ?)");
        $stmt->execute([$title, $description]);
        $survey_id = $pdo->lastInsertId();
        
        // Добавляем вопросы
        $questions = $_POST['questions'] ?? [];
        foreach ($questions as $index => $question_text) {
            if (!empty($question_text)) {
                $stmt = $pdo->prepare("
                    INSERT INTO survey_questions (survey_id, question_text, question_order) 
                    VALUES (?, ?, ?)
                ");
                $stmt->execute([$survey_id, $question_text, $index]);
                $question_id = $pdo->lastInsertId();
                
                // Добавляем варианты ответов
                $options = $_POST['options'][$index] ?? [];
                foreach ($options as $opt_index => $option_text) {
                    if (!empty($option_text)) {
                        $stmt = $pdo->prepare("
                            INSERT INTO question_options (question_id, option_text, option_order) 
                            VALUES (?, ?, ?)
                        ");
                        $stmt->execute([$question_id, $option_text, $opt_index]);
                    }
                }
            }
        }
        
        header("Location: survey-created.php?id=" . $survey_id);
        exit;
    }
}

    <script>
        let questionCount = 0;
        
        function addQuestion() {
            const container = document.getElementById('questions-container');
            const questionDiv = document.createElement('div');
            questionDiv.className = 'question-block';
            questionDiv.innerHTML = `
                <h3>Вопрос #${questionCount + 1}</h3>
                <input type="text" name="questions[]" placeholder="Текст вопроса" required style="width: 100%; padding: 10px; margin-bottom: 10px;">
                <button type="button" class="remove-btn" onclick="this.parentElement.remove()">Удалить вопрос</button>
                <div id="options-container-${questionCount}">
                    <h4>Варианты ответов:</h4>
                </div>
                <button type="button" class="add-btn" onclick="addOption(${questionCount})">+ Добавить вариант</button>
            `;
            container.appendChild(questionDiv);
            questionCount++;
        }
        
        function addOption(qNum) {
            const container = document.getElementById(`options-container-${qNum}`);
            const optionDiv = document.createElement('div');
            optionDiv.innerHTML = `
                <input type="text" name="options[${qNum}][]" class="option-input" placeholder="Вариант ответа" required>
                <button type="button" class="remove-btn" onclick="this.parentElement.remove()">X</button>
            `;
            container.appendChild(optionDiv);
        }
        
        // Добавляем первый вопрос при загрузке
        document.addEventListener('DOMContentLoaded', addQuestion);
