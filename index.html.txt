    <script>
        // Загрузка статистики при открытии страницы
        document.addEventListener('DOMContentLoaded', function() {
            loadQuickStats();
            loadPHPVersion();
        });

        // Загрузка быстрой статистики
        async function loadQuickStats() {
            try {
                const response = await fetch('api/stats.php');
                const data = await response.json();
                
                const statsContainer = document.getElementById('quickStats');
                statsContainer.innerHTML = `
                    <div class="stat-item">
                        <div class="stat-icon">
                            <i class="fas fa-users"></i>
                        </div>
                        <div class="stat-content">
                            <h3>Всего участников</h3>
                            <p class="stat-number">${data.total || 0}</p>
                        </div>
                    </div>
                    <div class="stat-item">
                        <div class="stat-icon">
                            <i class="fas fa-calendar-day"></i>
                        </div>
                        <div class="stat-content">
                            <h3>Ответов сегодня</h3>
                            <p class="stat-number">${data.today || 0}</p>
                        </div>
                    </div>
                    <div class="stat-item">
                        <div class="stat-icon">
                            <i class="fas fa-code"></i>
                        </div>
                        <div class="stat-content">
                            <h3>Популярная технология</h3>
                            <p class="stat-text">${data.top_tech || 'Нет данных'}</p>
                        </div>
                    </div>
                `;
            } catch (error) {
                console.error('Ошибка загрузки статистики:', error);
                document.getElementById('quickStats').innerHTML = `
                    <div class="error-message">
                        <i class="fas fa-exclamation-triangle"></i>
                        <p>Не удалось загрузить статистику</p>
                    </div>
                `;
            }
        }

        // Загрузка версии PHP
        async function loadPHPVersion() {
            try {
                const response = await fetch('api/info.php');
                const data = await response.json();
                document.getElementById('phpVersion').textContent = data.php_version;
            } catch (error) {
                document.getElementById('phpVersion').textContent = 'Ошибка загрузки';
            }
        }

        // Модальное окно "О проекте"
        function showAbout() {
            document.getElementById('aboutModal').style.display = 'block';
        }

        function closeAbout() {
            document.getElementById('aboutModal').style.display = 'none';
        }

        // Закрытие модального окна при клике вне его
        window.onclick = function(event) {
            const modal = document.getElementById('aboutModal');
            if (event.target == modal) {
                modal.style.display = 'none';
            }
        }
    </script>
